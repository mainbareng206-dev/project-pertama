    <script>
        const box = document.getElementById('box');
        const mysteryBox = document.getElementById('mysteryBox');
        const videoContainer = document.getElementById('videoContainer');
        const message = document.getElementById('message');
        const sparkles = document.getElementById('sparkles');
        let isOpened = false;

        mysteryBox.addEventListener('click', function() {
            if (isOpened) return;
            
            isOpened = true;
            
            // Buka box
            box.classList.add('open');
            
            // Tampilkan sparkles
            createSparkles();
            
            // Update message
            setTimeout(() => {
                message.textContent = 'WOW! Kejutan datang! 🎊';
            }, 500);
            
            // Tampilkan video
            setTimeout(() => {
                videoContainer.classList.add('show');
                message.textContent = 'Nikmati konten misteriusnya! 😍';
            }, 1500);
        });

        function createSparkles() {
            for (let i = 0; i < 20; i++) {
                setTimeout(() => {
                    const sparkle = document.createElement('div');
                    sparkle.className = 'sparkle';
                    sparkle.innerHTML = ['✨', '⭐', '🎇', '💫', '🌟'][Math.floor(Math.random() * 5)];
                    sparkle.style.left = Math.random() * 100 + '%';
                    sparkle.style.top = Math.random() * 100 + '%';
                    sparkle.style.animationDelay = Math.random() * 0.5 + 's';
                    sparkles.appendChild(sparkle);
                    
                    setTimeout(() => sparkle.remove(), 1000);
                }, i * 100);
            }
            sparkles.style.opacity = 1;
        }

        // Efek klik berulang
        mysteryBox.addEventListener('click', function() {
            if (isOpened) {
                mysteryBox.style.transform = 'scale(0.95)';
                setTimeout(() => {
                    mysteryBox.style.transform = 'scale(1.05)';
                }, 100);
            }
        });
    </script>
